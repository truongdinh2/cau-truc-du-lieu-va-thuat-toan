function infixToPostfix(input) { let stack = [], answer = [], char; const operands = { '+': 1, '-': 1, "*": 2, "/": 2, }; (let char of input.split(' ')) { let currentPrecedence = operands[char]; if (currentPrecedence) { let peek = operands[stack[stack.length - 1]]; // pop until the peek is smaller while (peek >= currentPrecedence) { answer.push(stack.pop()); peek = operands[stack[stack.length - 1]]; } stack.push(char); } else { // not operand, push to answer answer.push(char); } } while (stack.length > 0) { answer.push(stack.pop()) } return answer.join(' '); }
